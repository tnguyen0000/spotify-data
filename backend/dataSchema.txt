Types:
Artist = {
    id: string,
    name: string,
    spotifyUrl: string,
    imageUrl: string,
}

Term = {
    id: string,
    name: string,
    artists?: Artist[],
    spotifyUrl: string,
    imageUrl: string,
}

PlaylistStat = {
    stats [
        id?: string,
        name?: string,
        year?: int,
        genre?: string,
        count?: int,
        popularity?: int,
    ]
}


Collections:
topStatsArtist = {
    _id: string, # spotifyUserId
    expiry: time_in_seconds,
    type: "artist",
    shortTerm: Term[],
    mediumTerm: Term[],
    longTerm: Term[]
}
topStatsTrack = {
    _id: string, # spotifyUserId
    expiry: time_in_seconds,
    type: "track",
    shortTerm: Term[],
    mediumTerm: Term[],
    longTerm: Term[]
}

playListStats = {
    _id: string, # Spotify playlist ID
    expiry: time_in_seconds,
    stats: {
        fav_artist?: PlaylistStat[], 
        fav_genre?: PlaylistStat[], 
        fav_year?: PlaylistStat[], 
        popularity?: PlaylistStat[], 
    }
}
